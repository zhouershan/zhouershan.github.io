const int N = 1e6 + 10;
vector<int> vec[N], ans;
bool vis1[N], vis2[N];
int n, m;
int main() {
    read(n, m);
    for (int u, v, i = 1; i <= m; i++) read(u, v), vec[u].push_back(v);
    for (int i = 1; i <= n; i++)
        if (!vis1[i]) { vis1[i] = vis2[i] = 1; for (int v : vec[i]) vis1[v] = 1; }
    for (int i = n; i; i--)
        if (vis2[i]) { for (int v : vec[i]) vis2[v] = 0; ans.push_back(i); }
    println(ans.size());
    for (int v : ans) write(v, ' ');
    return 0;
}
